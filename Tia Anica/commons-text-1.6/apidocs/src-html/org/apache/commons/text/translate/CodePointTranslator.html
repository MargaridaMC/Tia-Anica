017package org.apache.commons.text.translate;
018
019import java.io.IOException;
020import java.io.Writer;
021
022/**
023 * Helper subclass to CharSequenceTranslator to allow for translations that
024 * will replace up to one character at a time.
025 *
026 * @since 1.0
027 */
028public abstract class CodePointTranslator extends CharSequenceTranslator {
029
030    /**
031     * Implementation of translate that maps onto the abstract translate(int, Writer) method.
032     * {@inheritDoc}
033     */
034    @Override
035    public final int translate(final CharSequence input, final int index, final Writer out) throws IOException {
036        final int codepoint = Character.codePointAt(input, index);
037        final boolean consumed = translate(codepoint, out);
038        return consumed ? 1 : 0;
039    }
040
041    /**
042     * Translate the specified codepoint into another.
043     *
044     * @param codepoint int character input to translate
045     * @param out Writer to optionally push the translated output to
046     * @return boolean as to whether translation occurred or not
047     * @throws IOException if and only if the Writer produces an IOException
048     */
049    public abstract boolean translate(int codepoint, Writer out) throws IOException;
050
051}

429    /**
430     * Used to invert an escape Map into an unescape Map.
431     * @param map Map&lt;String, String&gt; to be inverted
432     * @return Map&lt;String, String&gt; inverted array
433     */
434    public static Map<CharSequence, CharSequence> invert(final Map<CharSequence, CharSequence> map) {
435        final Map<CharSequence, CharSequence> newMap = new HashMap<>();
436        for (final Map.Entry<CharSequence, CharSequence> pair : map.entrySet()) {
437            newMap.put(pair.getValue(), pair.getKey());
438        }
439        return newMap;
440    }

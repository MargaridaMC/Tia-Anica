017package org.apache.commons.text;
018
019/**
020 * A predicate for selecting code points. Implementations of this interface must
021 * be thread safe.
022 *
023 * @since 1.0
024 */
025public interface CharacterPredicate {
026
027    /**
028     * Tests the code point with this predicate.
029     *
030     * @param codePoint
031     *            the code point to test
032     * @return {@code true} if the code point matches the predicate,
033     *         {@code false} otherwise
034     * @since 1.0
035     */
036    boolean test(int codePoint);
037}

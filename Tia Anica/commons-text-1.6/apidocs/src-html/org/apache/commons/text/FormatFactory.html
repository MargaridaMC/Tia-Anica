017package org.apache.commons.text;
018
019import java.text.Format;
020import java.util.Locale;
021
022/**
023 * Format factory.
024 *
025 * @since 1.0
026 */
027public interface FormatFactory {
028
029    /**
030     * Create or retrieve a format instance.
031     *
032     * @param name The format type name
033     * @param arguments Arguments used to create the format instance. This allows the
034     *                  <code>FormatFactory</code> to implement the "format style"
035     *                  concept from <code>java.text.MessageFormat</code>.
036     * @param locale The locale, may be null
037     * @return The format instance
038     */
039    Format getFormat(String name, String arguments, Locale locale);
040
041}
